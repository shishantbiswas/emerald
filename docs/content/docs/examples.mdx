# Examples

This page contains various examples showing different aspects of the Emerald programming language.

## Hello World

```emerald
function main(): i32 {
    print("Hello, World!");
    return 0;
}
```

## Variables and Types

```emerald
function main(): i32 {
    // Type inference
    let x = 42;           // i32
    let y = 3.14;         // f64
    let name = "Emerald"; // string
    let active = true;    // bool

    // Explicit typing
    let age: i32 = 25;

    print("x = ", x);
    print("y = ", y);
    print("name = ", name);
    print("active = ", active);
    print("age = ", age);

    return 0;
}
```

## Functions

```emerald
function factorial(n: i32): i32 {
    if n <= 1 {
        return 1;
    }
    return n * factorial(n - 1);
}

function fibonacci(n: i32): i32 {
    if n <= 1 {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

function main(): i32 {
    print("Factorial of 5: ", factorial(5));
    print("Fibonacci of 8: ", fibonacci(8));
    return 0;
}
```

## Control Flow

```emerald
function main(): i32 {
    let num = 15;

    // If-else chain
    if num % 15 == 0 {
        print(num, " is divisible by both 3 and 5");
    } else if num % 3 == 0 {
        print(num, " is divisible by 3");
    } else if num % 5 == 0 {
        print(num, " is divisible by 5");
    } else {
        print(num, " is not divisible by 3 or 5");
    }

    // While loop
    let i = 0;
    while i < 5 {
        print("Count: ", i);
        i = i + 1;
    }

    return 0;
}
```

## External Functions (FFI)

```emerald
// Declare external C function
extern function puts(str: *i8): i32;

// Use it in Emerald
function main(): i32 {
    puts("Hello from C!");
    return 0;
}
```
